var path = require('path'),
    fs = require('fs'),
    os = require('os'),
    test = require('unit.js'),
    loader = require('nsloader'),
    EntityCICore = loader('EntityCI/index'),
    Config = loader('EntityCI/Utils/Config');

describe('entity-ci/entity-ci-core/EntityCICore', function () {

  'use strict';

  var tmpPath = path.join(
        os.tmpdir(), 'entityci-tests--config--' + process.pid
      ),
      cfgFilename = path.join(tmpPath, 'config.json'),
      cfgData = {
        test: {value: 'hello'}
      };

  beforeEach(function () {

    fs.mkdirSync(tmpPath);
    fs.writeFileSync(cfgFilename, JSON.stringify(cfgData));

  });

  afterEach(function () {

      fs.unlinkSync(cfgFilename);
      fs.rmdirSync(tmpPath);

    });

  describe('EntityCICore', function () {

    it('loadConfigShouldBeAlmostPrivate', function () {

      var entityCICore = new EntityCICore(),
          loadConfig = Symbol.for('EntityCICore.loadConfig');

      test.value(
        entityCICore.loadConfig
      ).isUndefined();

      test.function(
        entityCICore[loadConfig]
      );

    });

    it('connectToMongoShouldBeAlmostPrivate', function () {

      var entityCICore = new EntityCICore(),
          connectToMongo = Symbol.for('EntityCICore.connectToMongo');

      test.value(
        entityCICore.connectToMongo
      ).isUndefined();

      test.function(
        entityCICore[connectToMongo]
      );

    });

  });

  describe('EntityCICore.loadConfig()', function () {

    it('shouldThrowAnErrorIfConfigDoesntExist', function (done) {

      var entityCICore = new EntityCICore(),
          sLoadConfig = Symbol.for('EntityCICore.loadConfig');

      entityCICore[sLoadConfig](cfgFilename, function (err) {

        test.error(
          err
        ).isInstanceOf(Error);

        done();

      });

    });

    it('shouldLoadConfig', function (done) {

      var entityCICore = new EntityCICore(),
          sLoadConfig = Symbol.for('EntityCICore.loadConfig');

      entityCICore[sLoadConfig](cfgFilename, function (err) {

        test.value(
          err
        ).isNull();

        test.object(
          entityCICore.config
        ).isInstanceOf(Config);

        test.object(
          entityCICore.config.data
        ).is(cfgData);

        done();

      });

    });

  });

  describe('EntityCICore.initialize()', function () {

    it('shouldInitializeWithoutIssues', function (done) {

      var entityCICore = new EntityCICore();

      test.bool(
        entityCICore.initialized
      ).isNotTrue();

      entityCICore.initialize(cfgFilename, function (err) {

        test.value(
          err
        ).isNull();

        test.bool(
          entityCICore.initialized
        ).isTrue();

        done();

      });

    });

  });

  describe('EntityCICore.config', function () {

    it('shouldReturnNullBeforeInitialize', function () {

      var entityCICore = new EntityCICore();

      test.value(
        entityCICore.config
      ).isNull();

    });

    it('shouldInitializeWithoutIssues', function (done) {

      var entityCICore = new EntityCICore();

      entityCICore.initialize('', function (err) {

        test.object(
          entityCICore.config
        ).isInstanceOf(Config);

        done();

      });

    });

  });

});
