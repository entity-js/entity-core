/**
 * Provides the EInvalidCharacters error which is thrown when the provided
 * value contains invalid characters.
 *
 * @author Orgun109uk <orgun109uk@gmail.com>
 *
 * @module entity-ci
 * @submodule entity-ci-core
 */

var sValue = Symbol('EInvalidCharacters.value'),
    sRule = Symbol('EInvalidCharacters.rule');

/**
 * Thrown when tryng to validate a value with invalid characters.
 *
 * @class EInvalidCharacters
 * @extends Error
 */
module.exports = class EInvalidCharacters extends Error {

  /**
   * The error constructor.
   *
   * @param {String} value The value being validated.
   * @param {String} rule The allowed characters.
   * @constructor
   */
  constructor(value, rule) {
    'use strict';

    super();

    this.stack = (new Error()).stack;
    this.name = this.constructor.name;
    this[sValue] = value;
    this[sRule] = rule;
  }

  /**
   * The value causing the error.
   *
   * @property value
   * @type String
   * @readOnly
   */
  get value() {
    'use strict';

    return this[sValue];
  }

  /**
   * The rule causing the error.
   *
   * @property rule
   * @type String
   * @readOnly
   */
  get rule() {
    'use strict';

    return this[sRule];
  }

};
