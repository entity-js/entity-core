/**
 * Provides the Entity validator rule.
 *
 * @author Orgun109uk <orgun109uk@gmail.com>
 *
 * @module entity-ci
 * @submodule entity-ci-core
 */

var loader = require('nsloader'),
    Entity = loader('EntityCI/EntityManager/Entity'),
    EInvalidEntity = loader('EntityCI/EntityManager/Errors/EInvalidEntity'),
    EFailedEntity = loader('EntityCI/EntityManager/Errors/EFailedEntity');

/**
 * Validate an Entity.
 *
 * @param {Mixed} value The value to validate.
 * @param {Object} options The options passed to the validator.
 * @param {Function} next The next callback.
 *   @param {Error} next.err Any raised errors.
 * @async
 *
 * @throws {EInvalidUrl} Thrown if the value is an invalid url address.
 */
module.exports = function validateEntity (value, options, next) {
  'use strict';

  if (value === undefined || value === null) {
    return next();
  }

  if (value instanceof Entity === false) {
    return next(new EInvalidEntity());
  }

  if (options) {
    for (var key in options) {
      if (value[key] !== options[key]) {
        return next(new EFailedEntity(key));
      }
    }
  }

  next();
};
