/**
 * Provides the EUnknownSchemaField error which is used when attempting to do
 * something with an unknown schema field.
 *
 * @author Orgun109uk <orgun109uk@gmail.com>
 *
 * @module entity-ci
 * @submodule entity-ci-core
 */

var sField = Symbol('EUnknownSchemaField.field');

/**
 * Thrown when do something with an unknown field.
 *
 * @class EUnknownSchemaField
 * @extends Error
 */
module.exports = class EUnknownSchemaField extends Error {

  /**
   * The error constructor.
   *
   * @param {String} field The name of the field.
   * @constructor
   */
  constructor(field) {
    'use strict';

    super();

    this.stack = (new Error()).stack;
    this.name = this.constructor.name;
    this[sField] = field;
  }

  /**
   * The schema field name.
   *
   * @property field
   * @type String
   * @readOnly
   */
  get field() {
    'use strict';

    return this[sField];
  }

};
