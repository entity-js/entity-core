/**
 * The entity ci core component.
 *
 * @author Orgun109uk <orgun109uk@gmail.com>
 */

var path = require('path'),
    async = require('async'),
    loader = require('nsloader'),
    Config = loader('EntityCI/Utils/Config'),
    Locale = loader('EntityCI/Locale'),
    Database = loader('EntityCI/Database'),
    Validators = loader('EntityCI/Validators'),
    Sanitizers = loader('EntityCI/Sanitizers'),
    EventManager = loader('EntityCI/EventManager'),
    PluginManager = loader('EntityCI/PluginManager'),
    EntityManager = loader('EntityCI/EntityManager');

var sConfig = Symbol('EntityCICore.config'),
    sLocale = Symbol('EntityCICore.locale'),
    sDatabase = Symbol('EntityCICore.database'),
    sValidators = Symbol('EntityCICore.validators'),
    sSanitizers = Symbol('EntityCICore.sanitizers'),
    sEventManager = Symbol('EntityCICore.eventManager'),
    sPluginManager = Symbol('EntityCICore.pluginManager'),
    sEntityManager = Symbol('EntityCICore.entityManager'),
    sIsInitialized = Symbol('EntityCICore.isInitialized'),
    sLoadConfig = Symbol.for('EntityCICore.loadConfig'),
    sConnectToMongo = Symbol.for('EntityCICore.connectToMongo'),
    sPlugins = Symbol.for('EntityCICore.plugins');

class EntityCICore {

  /**
   * The EntityCI Core class.
   */
  constructor() {
    'use strict';

    this[sIsInitialized] = false;
    this[sConfig] = null;
    this[sDatabase] = new Database();
    this[sValidators] = new Validators();
    this[sSanitizers] = new Sanitizers();
    this[sEventManager] = new EventManager();
    this[sPluginManager] = new PluginManager();
    this[sEntityManager] = new EntityManager(
      this[sDatabase],
      this[sValidators],
      this[sSanitizers]
    );
  }

  /**
   * Determine if the core has been initialized.
   *
   * @type Boolean
   * @default false
   * @readonly
   */
  get initialized() {
    'use strict';

    return this[sIsInitialized];
  }

  /**
   * Get the config object.
   *
   * @type Config
   * @readOnly
   */
  get config() {
    'use strict';

    return this[sConfig];
  }

  /**
   * Get the locale object.
   *
   * @type Locale
   * @readOnly
   */
  get locale() {
    'use strict';

    return this[sLocale];
  }

  /**
   * Get the database manager object.
   *
   * @type Database
   * @readOnly
   */
  get database() {
    'use strict';

    return this[sDatabase];
  }

  /**
   * Get the event manager object.
   *
   * @type EventManager
   * @readOnly
   */
  get eventManager() {
    'use strict';

    return this[sEventManager];
  }

  /**
   * Get the plugin manager object.
   *
   * @type PluginManager
   * @readOnly
   */
  get pluginManager() {
    'use strict';

    return this[sPluginManager];
  }

  /**
   * Load the configuration file.
   *
   * @name loadConfig
   * @function
   * @memberof EntityCICore
   * @param {String} filename The config filename.
   * @param {Function} done The done callback.
   *   @param {Error} done.err Any errors raised.
   */
  [sLoadConfig](filename, done) {
    'use strict';

    this[sConfig] = new Config(filename);
    this[sConfig].load(done);
  }

  /**
   * Connects to the MongoDB, details provided by the config.
   *
   * @function connectToMongo
   * @param {Function} done The done callback.
   *   @param {Error} done.err Any errors raised.
   * @private
   */
  [sConnectToMongo](done) {
    'use strict';

    var me = this,
        queue = [];

    function connect(name) {
      return function (next) {
        me[sDatabase].connect(
          name,
          me[sConfig].get('database.' + name, {
            host: '0.0.0.0',
            name: 'entityci'
          })
        );

        next();
      };
    }

    // @todo - multiple connections?
    queue.push(connect('default'));

    async.series(queue, function (err) {
      done(err ? err : null);
    });
  }

  /**
   * Indexes and initializes the plugin manager and enabled plugins.
   *
   * @name plugins
   * @memberof EntityCICore
   * @param {Function} done The done callback.
   *   @param {Error} done.err Any errors raised.
   * @private
   */
  [sPlugins](done) {
    'use strict';

    var me = this,
        queue = [];

    queue.push(function (next) {
      me[sPluginManager].index(
        me[sConfig].get(
          'plugins.path',
          path.join(__dirname, '..', 'plugins')
        ),
        next
      );
    });

    queue.push(function (next) {
      me[sPluginManager].initialize(
        me[sConfig].get('plugins.enabled', []),
        next
      );
    });

    async.series(queue, function (err) {
      done(err ? err : null);
    });
  }

  /**
   * Initializes the core application.
   *
   * @memberof EntityCICore
   * @param {String} filename The filename of the config file to load.
   * @param {Function} done The done callback.
   *   @param {Error} done.err Any errors raised.
   */
  initialize(filename, done) {
    'use strict';

    var me = this,
        queue = [];

    queue.push(function (next) {
      me[sLoadConfig](filename, next);
    });

    queue.push(function (next) {
      me[sConnectToMongo](next);
    });

    queue.push(function (next) {
      me[sLocale] = new Locale(me[sDatabase]);
      me[sLocale].initialize(me[sConfig].get('locale.dir'), next);
    });

    queue.push(function (next) {
      me[sPlugins](next);
    });

    queue.push(function (next) {
      me[sEventManager].fire('initialized', next, {
        core: me
      });
    });

    async.series(queue, function (err) {
      if (!err) {
        me[sIsInitialized] = true;
      }

      done(err ? err : null);
    });
  }

}

module.exports = EntityCICore;
