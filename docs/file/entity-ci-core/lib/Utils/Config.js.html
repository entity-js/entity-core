/**
 * Provides the Config class allowing access to the config.
 *
 * @author Orgun109uk <orgun109uk@gmail.com>
 *
 * @module entity-ci
 * @submodule entity-ci-core
 */

var fs = require('fs'),
    loader = require('nsloader'),
    Data = loader('EntityCI/Utils/Data');

var sConfigFilename = Symbol('Config.filename');

/**
 * The Config class providing methods to access the config options.
 *
 * @class Config
 */
module.exports = class Config extends Data {

  /**
   * Construct the config object.
   *
   * @param {String} filename The filename of the config file.
   * @constructor
   */
  constructor(filename) {
    'use strict';

    super(null, '.');
    this[sConfigFilename] = filename;
  }

  /**
   * @todo
   */
  get filename() {
    return this[sConfigFilename];
  }

  /**
   * Saves the config file.
   *
   * @method save
   * @param {Function} done The done callback.
   *   @param {Error} done.err Any errors raised.
   * @async
   */
  save(done) {
    'use strict';

    fs.writeFile(
      this.filename,
      JSON.stringify(this.data),
      function (err) {
        done(err ? err : null);
      }
    );
  }

  /**
   * Restores the config object from the config file.
   *
   * @method load
   * @param {Function} done The done callback.
   *   @param {Error} done.err Any raised errors.
   */
  load(done) {
    'use strict';

    if (this.filename === '') {
      return done(new Error());
    }

    var me = this;
    fs.readFile(this.filename, function (err, data) {
      if (err) {
        return done(err);
      }

      try {
        me.data = JSON.parse(data);
        done(null);
      } catch (e) {
        done(e);
      }
    });
  }
};
