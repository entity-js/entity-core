require('babel/register');

var path = require('path'),
    fs = require('fs'),
    loader = require('nsloader');

// Allow access to the EntityCI index file.
loader.register('EntityCI', function () {
  'use strict';

  return path.resolve(path.join(__dirname, 'index.js'));
});

// Allow access to anything after 'EntityCI'.
loader.register('EntityCI/*', function (ns) {
  'use strict';

  var filename = path.join(__dirname, 'lib', ns.substring('EntityCI/'.length));
  if (fs.existsSync(filename + '.js')) {
    return filename + '.js';
  } else if (fs.existsSync(path.join(filename, 'index.js'))) {
    return path.join(filename, 'index.js');
  }

  return false;
});

var EntityCICore = require('./lib');
global.entityCICore = module.exports = new EntityCICore();
