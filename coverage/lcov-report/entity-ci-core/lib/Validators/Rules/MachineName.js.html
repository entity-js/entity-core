function validateMachineName(value, options, next) {
  'use strict';
 
  var min = 3,
      max = 128;
 
  if (value.length < min || value.length > max) {
    return next(new EInvalidLength(value, min, max));
  }
 
  if (value.match(/[^-_a-z0-9]/) !== null) {

  next();
}
